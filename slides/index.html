:valid & :invalid

                <form class="form">
                  <h1>Регистрация</h1>
                  <fieldset>
                      <input placeholder="Введите имя" type="text" required class="form__input">
                      <input placeholder="Введите почту" type="email" required class="form__input">
                      <input placeholder="Введите пароль" type="password" required class="form__input"
            minlength="5"
        >
                      <button type="submit" class="form__submit">Зарегистрироваться</button>
                  </fieldset>
                </form>

:valid & :invalid

            .form__input:required {
              border-color: grey;
              background-image: url('./images/required.svg');
            }
        
            .form__input:required:not(:placeholder-shown):focus:invalid {
              border-color: red;
              background-image: url('./images/required-invalid.svg');
            }
        
            .form__input:required:valid {
              border-color: green;
              background-image: url('./images/required-valid.svg');
            }

:valid & :invalid

            // Если любое из полей invalid скрываем кнопку регистрации
            .form fieldset:invalid .form__submit {
              display: none;
            }

:valid & :invalid

                // Ошибка длины пароля
                .form .form__error {
                  display: none;
                  color: red;
                }
            
                // Если placeholder не показывается и поле invalid
                .form .form__input[type="password"]:not(:placeholder-shown):invalid + .form__error {
                  display: block;
                }
